# LocalScribe v2.0 - Dependencies
# Install: pip install -r requirements.txt
#
# Note for corp Macs with TLS interception:
#   configure your organization CA for pip (PIP_CERT or global.cert).
#   Do not bypass certificate verification with --trusted-host.
#   On macOS, the system bundle is usually: /etc/ssl/cert.pem

# Transcription (Hebrew ASR on Apple Silicon)
mlx-whisper>=0.4.0

# Speaker Diarization
# pyannote.audio 4.x requires an additional gated `speaker-diarization-community-1` model
# and changes the public API. Pin to 3.x to keep the README install flow working.
pyannote.audio>=3.1.0,<4.0

# Torch 2.6 changed the default of `torch.load(weights_only=True)`, breaking pyannote
# checkpoints. Torch 2.7+ also drops `torchaudio.AudioMetaData`. Pin to 2.5.x.
torch>=2.2,<2.6
torchaudio>=2.2,<2.6

# huggingface_hub 1.x removed `use_auth_token` (which pyannote 3.x still uses).
huggingface_hub<0.30

# Audio processing
# pydub is no longer used directly (some Mac endpoint security tools — e.g. Santa —
# kill ffprobe when launched as a Python subprocess). We use ffmpeg + soundfile instead.
soundfile>=0.12

# Document processing
pdfplumber>=0.10.0
python-docx>=1.0.0
